#### 1.5 ‡∏™‡∏£‡πâ‡∏≤‡∏á MCP Server ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£‡πÑ‡∏ü‡∏•‡πå CSV
‡∏™‡∏£‡πâ‡∏≤‡∏á‡πÑ‡∏ü‡∏•‡πå `server.py` ‡πÉ‡∏ô‡πÇ‡∏ü‡∏•‡πÄ‡∏î‡∏≠‡∏£‡πå‡πÇ‡∏õ‡∏£‡πÄ‡∏à‡∏Å‡∏ï‡πå‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏∏‡∏ì ‡πÅ‡∏•‡∏∞‡πÄ‡∏û‡∏¥‡πà‡∏°‡πÇ‡∏Ñ‡πâ‡∏î‡∏ï‡πà‡∏≠‡πÑ‡∏õ‡∏ô‡∏µ‡πâ‡∏•‡∏á‡πÑ‡∏õ:
```python
import logging
import sys
from fastmcp import FastMCP
import pandas as pd
from pathlib import Path

# ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ Logging ‡πÉ‡∏´‡πâ‡πÅ‡∏™‡∏î‡∏á‡∏ú‡∏•‡∏ó‡∏µ‡πà stderr
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
    stream=sys.stderr,
)
logger = logging.getLogger(__name__)

# Initialize FastMCP instance
mcp = FastMCP("csv-tools")

# Define the data directory path
DATA_DIR = Path(__file__).parent / "data"


# --- Helper Function: ‡πÅ‡∏õ‡∏•‡∏á NaN ‡πÄ‡∏õ‡πá‡∏ô None ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ JSON ‡πÑ‡∏°‡πà‡∏û‡∏±‡∏á ---
def clean_nans(data):
    """‡πÅ‡∏õ‡∏•‡∏á‡∏Ñ‡πà‡∏≤ NaN ‡πÉ‡∏ô DataFrame/Series ‡πÉ‡∏´‡πâ‡πÄ‡∏õ‡πá‡∏ô None ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ‡∏™‡πà‡∏á JSON ‡πÑ‡∏î‡πâ‡πÑ‡∏°‡πà error"""
    if isinstance(data, (pd.DataFrame, pd.Series)):
        return data.where(pd.notnull(data), None)
    return data


# Tool to list all CSV files in the data directory
@mcp.tool()
def list_csv_files() -> list[str]:
    """‡πÅ‡∏™‡∏î‡∏á‡∏£‡∏≤‡∏¢‡∏ä‡∏∑‡πà‡∏≠‡πÑ‡∏ü‡∏•‡πå CSV ‡∏ó‡∏µ‡πà‡∏û‡∏£‡πâ‡∏≠‡∏°‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô‡πÉ‡∏ô‡πÇ‡∏ü‡∏•‡πÄ‡∏î‡∏≠‡∏£‡πå data"""
    logger.info(f"‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏•‡∏¥‡∏™‡∏ï‡πå‡πÑ‡∏ü‡∏•‡πå CSV ‡πÉ‡∏ô‡πÇ‡∏ü‡∏•‡πÄ‡∏î‡∏≠‡∏£‡πå: {DATA_DIR}")
    # ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤‡πÇ‡∏ü‡∏•‡πÄ‡∏î‡∏≠‡∏£‡πå‡∏°‡∏µ‡∏≠‡∏¢‡∏π‡πà‡∏à‡∏£‡∏¥‡∏á‡∏Å‡πà‡∏≠‡∏ô glob
    if not DATA_DIR.exists():
        return []
    return [p.name for p in DATA_DIR.glob("*.csv")]


# Tool to summarize a specific CSV file
@mcp.tool()
def summarize_csv(file_name: str) -> dict:
    """
    ‡∏™‡∏£‡∏∏‡∏õ‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÑ‡∏ü‡∏•‡πå CSV (‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡πÅ‡∏ñ‡∏ß, ‡∏Ñ‡∏≠‡∏•‡∏±‡∏°‡∏ô‡πå, ‡∏ï‡∏±‡∏ß‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏´‡∏±‡∏ß‡∏ï‡∏≤‡∏£‡∏≤‡∏á, ‡∏™‡∏ñ‡∏¥‡∏ï‡∏¥‡∏£‡∏ß‡∏°‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏≠‡∏•‡∏±‡∏°‡∏ô‡πå‡∏ï‡∏±‡∏ß‡πÄ‡∏•‡∏Ç)
    """
    logger.info(f"‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏™‡∏£‡∏∏‡∏õ‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å‡πÑ‡∏ü‡∏•‡πå: {file_name}")
    file_path = DATA_DIR / file_name

    # ‡πÄ‡∏ä‡πá‡∏Ñ‡∏ß‡πà‡∏≤‡∏°‡∏µ‡πÑ‡∏ü‡∏•‡πå‡∏à‡∏£‡∏¥‡∏á‡πÑ‡∏´‡∏°
    if not file_path.exists():
        raise FileNotFoundError(f"‡πÑ‡∏°‡πà‡∏û‡∏ö‡πÑ‡∏ü‡∏•‡πå {file_name} ‡πÉ‡∏ô‡πÇ‡∏ü‡∏•‡πÄ‡∏î‡∏≠‡∏£‡πå data")

    df = pd.read_csv(file_path)

    return {
        "file": file_name,
        "shape": {"rows": int(df.shape[0]), "cols": int(df.shape[1])},
        "columns": list(df.columns),
        "head": clean_nans(df.head(5)).to_dict(orient="records"),
        "numeric_summary": clean_nans(df.describe(include="number")).to_dict(),
    }


# Tool to read rows with pagination
@mcp.tool()
def read_csv_rows(file_name: str, offset: int = 0, limit: int = 10) -> list[dict]:
    """‡∏≠‡πà‡∏≤‡∏ô‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å‡πÑ‡∏ü‡∏•‡πå CSV ‡πÅ‡∏ö‡∏ö‡∏£‡∏∞‡∏ö‡∏∏‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡πÅ‡∏ñ‡∏ß (Pagination)"""
    logger.info(f"‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏≠‡πà‡∏≤‡∏ô‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å‡πÑ‡∏ü‡∏•‡πå: {file_name}, offset={offset}, limit={limit}")
    file_path = DATA_DIR / file_name

    if not file_path.exists():
        raise FileNotFoundError(f"‡πÑ‡∏°‡πà‡∏û‡∏ö‡πÑ‡∏ü‡∏•‡πå {file_name}")

    df = pd.read_csv(file_path)

    if offset >= len(df):
        return []

    # ‡∏ï‡∏±‡∏î‡πÅ‡∏ñ‡∏ß‡∏ó‡∏µ‡πà‡∏ï‡πâ‡∏≠‡∏á‡∏Å‡∏≤‡∏£‡πÅ‡∏•‡∏∞ clean ‡∏Ñ‡πà‡∏≤‡∏ß‡πà‡∏≤‡∏á
    result_df = clean_nans(df.iloc[offset : offset + limit])
    return result_df.to_dict(orient="records")


# Tool to get unique values
@mcp.tool()
def get_unique_values(file_name: str, column_name: str) -> list:
    """‡∏î‡∏∂‡∏á‡∏Ñ‡πà‡∏≤‡∏ó‡∏µ‡πà‡πÑ‡∏°‡πà‡∏ã‡πâ‡∏≥‡∏Å‡∏±‡∏ô (Unique values) ‡∏à‡∏≤‡∏Å‡∏Ñ‡∏≠‡∏•‡∏±‡∏°‡∏ô‡πå‡∏ó‡∏µ‡πà‡∏£‡∏∞‡∏ö‡∏∏"""
    logger.info(f"‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏î‡∏∂‡∏á‡∏Ñ‡πà‡∏≤‡∏ó‡∏µ‡πà‡πÑ‡∏°‡πà‡∏ã‡πâ‡∏≥‡∏Å‡∏±‡∏ô‡∏à‡∏≤‡∏Å‡πÑ‡∏ü‡∏•‡πå: {file_name}, ‡∏Ñ‡∏≠‡∏•‡∏±‡∏°‡∏ô‡πå: {column_name}")
    file_path = DATA_DIR / file_name

    if not file_path.exists():
        raise FileNotFoundError(f"‡πÑ‡∏°‡πà‡∏û‡∏ö‡πÑ‡∏ü‡∏•‡πå {file_name}")

    df = pd.read_csv(file_path)

    if column_name not in df.columns:
        raise ValueError(f"‡πÑ‡∏°‡πà‡∏û‡∏ö‡∏Ñ‡∏≠‡∏•‡∏±‡∏°‡∏ô‡πå {column_name}")

    return df[column_name].dropna().unique().tolist()


# Tool to filter data using pandas query
@mcp.tool()
def filter_csv(file_name: str, query: str) -> list[dict]:
    """
    ‡∏Å‡∏£‡∏≠‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• CSV ‡∏î‡πâ‡∏ß‡∏¢ query string (‡πÉ‡∏ä‡πâ syntax ‡∏Ç‡∏≠‡∏á pandas query)

    ‡∏ï‡∏±‡∏ß‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏Å‡∏≤‡∏£‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô:
    - "age > 25"
    - "department == 'Sales' and salary > 50000"
    - "product_name.str.contains('Pro', case=False)"
    """
    logger.info(f"‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏Å‡∏£‡∏≠‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å‡πÑ‡∏ü‡∏•‡πå: {file_name} ‡∏î‡πâ‡∏ß‡∏¢ query: {query}")
    file_path = DATA_DIR / file_name

    if not file_path.exists():
        raise FileNotFoundError(f"‡πÑ‡∏°‡πà‡∏û‡∏ö‡πÑ‡∏ü‡∏•‡πå {file_name}")

    df = pd.read_csv(file_path)

    try:
        result = df.query(query, engine="python")
        return clean_nans(result).to_dict(orient="records")
    except Exception as e:
        raise ValueError(f"Query Error: {str(e)}")


if __name__ == "__main__":
    # Run the MCP server with SSE transport on port 8000
    mcp.run(transport="sse", host="0.0.0.0", port=8000)
```

```python
LLM_API_KEY = os.getenv("LLM_API_KEY")

# -----------------------------
def call_llm_for_summary(
    content: str,
    extra_instruction: Optional[str] = None,
    max_tokens: int = 1024,
) -> str:
    """
    ‡πÄ‡∏£‡∏µ‡∏¢‡∏Å LLM ‡∏ú‡πà‡∏≤‡∏ô API ‡∏£‡∏π‡∏õ‡πÅ‡∏ö‡∏ö OpenAI-compatible ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏™‡∏£‡∏∏‡∏õ/‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå‡πÄ‡∏≠‡∏Å‡∏™‡∏≤‡∏£
    """
    if not LLM_API_KEY:
        raise RuntimeError("‡∏Å‡∏£‡∏∏‡∏ì‡∏≤‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ LLM_API_KEY ‡πÉ‡∏ô environment ‡∏Å‡πà‡∏≠‡∏ô‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô")

    system_prompt = (
        "‡∏Ñ‡∏∏‡∏ì‡∏Ñ‡∏∑‡∏≠‡∏ú‡∏π‡πâ‡∏ä‡πà‡∏ß‡∏¢‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå‡πÄ‡∏≠‡∏Å‡∏™‡∏≤‡∏£‡∏£‡∏∞‡∏î‡∏±‡∏ö‡∏°‡∏∑‡∏≠‡∏≠‡∏≤‡∏ä‡∏µ‡∏û "
        "‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏Å‡∏±‡∏ö‡πÄ‡∏≠‡∏Å‡∏™‡∏≤‡∏£‡∏†‡∏≤‡∏©‡∏≤‡πÑ‡∏ó‡∏¢‡πÅ‡∏•‡∏∞‡∏≠‡∏±‡∏á‡∏Å‡∏§‡∏©‡πÑ‡∏î‡πâ‡∏î‡∏µ "
        "‡πÉ‡∏´‡πâ‡∏™‡∏£‡∏∏‡∏õ‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏ä‡∏±‡∏î‡πÄ‡∏à‡∏ô ‡πÄ‡∏Ç‡πâ‡∏≤‡πÉ‡∏à‡∏á‡πà‡∏≤‡∏¢ ‡πÅ‡∏•‡∏∞‡∏î‡∏∂‡∏á insight ‡∏ó‡∏µ‡πà‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç‡∏≠‡∏≠‡∏Å‡∏°‡∏≤"
    )

    if extra_instruction:
        user_prompt = (
            f"{extra_instruction}\n\n" "----- ‡πÄ‡∏ô‡∏∑‡πâ‡∏≠‡∏´‡∏≤‡∏à‡∏≤‡∏Å OCR -----\n" f"{content}\n"
        )
    else:
        user_prompt = (
            "‡∏Å‡∏£‡∏∏‡∏ì‡∏≤‡∏™‡∏£‡∏∏‡∏õ‡πÄ‡∏ô‡∏∑‡πâ‡∏≠‡∏´‡∏≤ ‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå‡∏õ‡∏£‡∏∞‡πÄ‡∏î‡πá‡∏ô‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç "
            "‡πÅ‡∏•‡∏∞‡∏î‡∏∂‡∏á insight ‡∏ó‡∏µ‡πà‡∏ô‡πà‡∏≤‡∏™‡∏ô‡πÉ‡∏à‡∏à‡∏≤‡∏Å‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏î‡πâ‡∏≤‡∏ô‡∏•‡πà‡∏≤‡∏á:\n\n"
            "----- ‡πÄ‡∏ô‡∏∑‡πâ‡∏≠‡∏´‡∏≤‡∏à‡∏≤‡∏Å OCR -----\n"
            f"{content}\n"
        )

    url = f"{LLM_BASE_URL.rstrip('/')}/chat/completions"

    headers = {
        "Authorization": f"Bearer {LLM_API_KEY}",
        "Content-Type": "application/json",
    }

    # ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤‡πÄ‡∏õ‡πá‡∏ô Model ‡∏ï‡∏£‡∏∞‡∏Å‡∏π‡∏• o1/o3 ‡∏´‡∏£‡∏∑‡∏≠‡πÑ‡∏°‡πà (‡∏ó‡∏µ‡πà‡∏ö‡∏±‡∏á‡∏Ñ‡∏±‡∏ö‡πÉ‡∏ä‡πâ max_completion_tokens)
    # ‡∏´‡∏£‡∏∑‡∏≠‡∏ñ‡πâ‡∏≤ API ‡πÅ‡∏à‡πâ‡∏á error ‡∏ô‡∏µ‡πâ‡∏°‡∏≤ ‡∏Å‡πá‡∏Ñ‡∏ß‡∏£‡πÄ‡∏õ‡∏•‡∏µ‡πà‡∏¢‡∏ô‡πÑ‡∏õ‡πÉ‡∏ä‡πâ max_completion_tokens
    # ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡∏á‡πà‡∏≤‡∏¢ ‡πÄ‡∏£‡∏≤‡∏à‡∏∞‡πÄ‡∏ä‡πá‡∏Ñ‡∏à‡∏≤‡∏Å‡∏ä‡∏∑‡πà‡∏≠ Model ‡∏´‡∏£‡∏∑‡∏≠‡πÉ‡∏ä‡πâ try-except ‡∏Å‡πá‡πÑ‡∏î‡πâ
    # ‡πÅ‡∏ï‡πà‡πÉ‡∏ô‡∏ó‡∏µ‡πà‡∏ô‡∏µ‡πâ‡∏Ç‡∏≠‡∏õ‡∏£‡∏±‡∏ö‡πÉ‡∏´‡πâ‡∏£‡∏≠‡∏á‡∏£‡∏±‡∏ö‡∏ó‡∏±‡πâ‡∏á‡∏Ñ‡∏π‡πà‡πÇ‡∏î‡∏¢‡∏î‡∏π‡∏à‡∏≤‡∏Å‡∏ä‡∏∑‡πà‡∏≠ Model ‡∏Ñ‡∏£‡πà‡∏≤‡∏ß‡πÜ ‡∏´‡∏£‡∏∑‡∏≠ default ‡πÄ‡∏õ‡πá‡∏ô max_tokens

    is_reasoning_model = any(
        x in LLM_MODEL.lower() for x in ["o1", "o3", "gpt-5"]
    )  # ‡πÄ‡∏î‡∏≤‡∏ß‡πà‡∏≤ gpt-5 ‡∏≠‡∏≤‡∏à‡∏à‡∏∞‡πÄ‡∏õ‡πá‡∏ô reasoning

    messages = []
    if is_reasoning_model:

                "role": "user",

        ]

    }

            f"LLM API Error (400 Bad Request): {error_detail}. Please check your LLM_MODEL or API Key."

    resp.raise_for_status()
    data = resp.json()

    return ""

# Helper Function: ‡πÅ‡∏õ‡∏•‡∏á PDF ‡∏´‡∏ô‡πâ‡∏≤‡∏ó‡∏µ‡πà‡∏£‡∏∞‡∏ö‡∏∏‡πÄ‡∏õ‡πá‡∏ô‡∏£‡∏π‡∏õ‡∏†‡∏≤‡∏û (‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÄ‡∏•‡∏µ‡πà‡∏¢‡∏á Poppler)
# -----------------------------
def convert_pdf_page_to_image(pdf_path: Path, page_num: int) -> str:
    """
    Convert a specific page of a PDF to a temporary image file using PyMuPDF.
    Returns the path to the temporary image file.
    """
    doc = fitz.open(pdf_path)
    try:
        # page_num is 1-based from user, fitz is 0-based
        if page_num < 1 or page_num > len(doc):
            raise ValueError(f"Page number {page_num} is out of range (1-{len(doc)})")

        page = doc.load_page(page_num - 1)
        pix = page.get_pixmap(dpi=300)  # 300 DPI for good OCR

        # Create a temp file
        fd, temp_path = tempfile.mkstemp(suffix=".png")
        os.close(fd)

        pix.save(temp_path)
        return temp_path
    finally:
        doc.close()

# -----------------------------
def _ocr_logic(file_name: str, page_num: int) -> str:
    if not TYPHOON_OCR_API_KEY:
        raise RuntimeError(
            "‡∏Å‡∏£‡∏∏‡∏ì‡∏≤‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ TYPHOON_OCR_API_KEY ‡πÉ‡∏ô environment ‡∏Å‡πà‡∏≠‡∏ô‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô Typhoon OCR"
        )

    file_path = PDF_DIR / file_name

    # ‡πÉ‡∏ä‡πâ PyMuPDF ‡πÅ‡∏õ‡∏•‡∏á‡πÄ‡∏õ‡πá‡∏ô‡∏£‡∏π‡∏õ‡∏†‡∏≤‡∏û‡∏Å‡πà‡∏≠‡∏ô ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏´‡∏•‡∏µ‡∏Å‡πÄ‡∏•‡∏µ‡πà‡∏¢‡∏á‡∏õ‡∏±‡∏ç‡∏´‡∏≤ Poppler missing ‡∏ö‡∏ô Windows
    temp_image_path = convert_pdf_page_to_image(file_path, page_num)

    try:
        # ‡∏™‡πà‡∏á‡∏£‡∏π‡∏õ‡∏†‡∏≤‡∏û‡πÑ‡∏õ‡πÉ‡∏´‡πâ Typhoon OCR ‡πÅ‡∏ó‡∏ô‡πÑ‡∏ü‡∏•‡πå PDF
        markdown = ocr_document(
            pdf_or_image_path=temp_image_path,
            # page_num ‡πÑ‡∏°‡πà‡∏à‡∏≥‡πÄ‡∏õ‡πá‡∏ô‡∏ï‡πâ‡∏≠‡∏á‡∏£‡∏∞‡∏ö‡∏∏‡πÄ‡∏°‡∏∑‡πà‡∏≠‡∏™‡πà‡∏á‡πÄ‡∏õ‡πá‡∏ô‡∏£‡∏π‡∏õ‡∏†‡∏≤‡∏û‡πÄ‡∏î‡∏µ‡πà‡∏¢‡∏ß
            api_key=TYPHOON_OCR_API_KEY,
        )
    finally:
        # ‡∏•‡∏ö‡πÑ‡∏ü‡∏•‡πå‡∏£‡∏π‡∏õ‡∏†‡∏≤‡∏û‡∏ä‡∏±‡πà‡∏ß‡∏Ñ‡∏£‡∏≤‡∏ß
        if os.path.exists(temp_image_path):
            os.remove(temp_image_path)

    return markdown

    """
    ‡πÅ‡∏™‡∏î‡∏á‡∏£‡∏≤‡∏¢‡∏ä‡∏∑‡πà‡∏≠‡πÑ‡∏ü‡∏•‡πå PDF ‡∏ó‡∏µ‡πà‡∏°‡∏µ‡πÉ‡∏ô‡πÇ‡∏ü‡∏•‡πÄ‡∏î‡∏≠‡∏£‡πå data/pdf

    """

    - file_name: ‡∏ä‡∏∑‡πà‡∏≠‡πÑ‡∏ü‡∏•‡πå PDF ‡πÉ‡∏ô data/pdf
    - page_num: ‡πÄ‡∏•‡∏Ç‡∏´‡∏ô‡πâ‡∏≤‡∏ó‡∏µ‡πà‡∏ï‡πâ‡∏≠‡∏á‡∏Å‡∏≤‡∏£ OCR (‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏à‡∏≤‡∏Å 1)
    - extra_instruction: ‡∏ñ‡πâ‡∏≤‡∏≠‡∏¢‡∏≤‡∏Å‡∏Å‡∏≥‡∏´‡∏ô‡∏î‡∏£‡∏π‡∏õ‡πÅ‡∏ö‡∏ö‡∏™‡∏£‡∏∏‡∏õ ‡πÄ‡∏ä‡πà‡∏ô
        "‡∏ä‡πà‡∏ß‡∏¢‡∏™‡∏£‡∏∏‡∏õ‡πÄ‡∏õ‡πá‡∏ô bullet 5 ‡∏Ç‡πâ‡∏≠‡πÅ‡∏•‡∏∞‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå‡πÅ‡∏ô‡∏ß‡πÇ‡∏ô‡πâ‡∏°‡∏¢‡∏≠‡∏î‡∏Ç‡∏≤‡∏¢"
    """
    # 1) OCR ‡∏î‡πâ‡∏ß‡∏¢ Typhoon OCR
    markdown = _ocr_logic(file_name=file_name, page_num=page_num)

    # ‡∏ñ‡πâ‡∏≤‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡∏¢‡∏≤‡∏ß‡∏°‡∏≤‡∏Å ‡∏≠‡∏≤‡∏à‡∏ï‡∏±‡∏î‡∏Ñ‡∏ß‡∏≤‡∏°‡∏¢‡∏≤‡∏ß‡∏Å‡πà‡∏≠‡∏ô‡∏™‡πà‡∏á‡πÄ‡∏Ç‡πâ‡∏≤ LLM

    else:
        markdown_to_use = markdown

    # 2) ‡∏™‡πà‡∏á‡πÄ‡∏Ç‡πâ‡∏≤ LLM ‡πÉ‡∏´‡πâ‡∏™‡∏£‡∏∏‡∏õ / ‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå
    summary = call_llm_for_summary(
        content=markdown_to_use,
        extra_instruction=extra_instruction,
        max_tokens=1024,
    )

    return summary


@mcp.tool()
def ocr_pdf_multi_page_and_summarize(
    file_name: str,
    start_page: int = 1,
    end_page: int = 3,
    extra_instruction: Optional[str] = None,
) -> str:
    """
    OCR PDF ‡∏´‡∏•‡∏≤‡∏¢‡∏´‡∏ô‡πâ‡∏≤ (‡∏ä‡πà‡∏ß‡∏á start_page ‡∏ñ‡∏∂‡∏á end_page) ‡πÅ‡∏•‡πâ‡∏ß‡πÉ‡∏´‡πâ LLM ‡∏™‡∏£‡∏∏‡∏õ/‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå‡∏£‡∏ß‡∏°
    ‡πÉ‡∏ä‡πâ‡∏Å‡∏£‡∏ì‡∏µ‡∏£‡∏≤‡∏¢‡∏á‡∏≤‡∏ô‡∏´‡∏•‡∏≤‡∏¢‡∏´‡∏ô‡πâ‡∏≤ ‡πÄ‡∏ä‡πà‡∏ô Report ‡∏ó‡∏±‡πâ‡∏á‡πÄ‡∏î‡∏∑‡∏≠‡∏ô/‡∏ó‡∏±‡πâ‡∏á‡πÑ‡∏ï‡∏£‡∏°‡∏≤‡∏™
    """
    if start_page < 1:
        raise ValueError("start_page ‡∏ï‡πâ‡∏≠‡∏á‡∏°‡∏≤‡∏Å‡∏Å‡∏ß‡πà‡∏≤‡∏´‡∏£‡∏∑‡∏≠‡πÄ‡∏ó‡πà‡∏≤‡∏Å‡∏±‡∏ö 1")
    if end_page < start_page:
        raise ValueError("end_page ‡∏ï‡πâ‡∏≠‡∏á‡∏°‡∏≤‡∏Å‡∏Å‡∏ß‡πà‡∏≤‡∏´‡∏£‡∏∑‡∏≠‡πÄ‡∏ó‡πà‡∏≤‡∏Å‡∏±‡∏ö start_page")

    # OCR ‡∏ó‡∏µ‡∏•‡∏∞‡∏´‡∏ô‡πâ‡∏≤‡πÅ‡∏•‡πâ‡∏ß‡∏ï‡πà‡∏≠‡πÄ‡∏õ‡πá‡∏ô‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡πÄ‡∏î‡∏µ‡∏¢‡∏ß
    combined_markdown_parts: list[str] = []
    for page in range(start_page, end_page + 1):
        try:
            page_md = _ocr_logic(file_name=file_name, page_num=page)
            combined_markdown_parts.append(f"# ‡∏´‡∏ô‡πâ‡∏≤ {page}\n\n{page_md}")
        except Exception as e:
            combined_markdown_parts.append(f"# ‡∏´‡∏ô‡πâ‡∏≤ {page}\n\n(OCR ‡∏ú‡∏¥‡∏î‡∏û‡∏•‡∏≤‡∏î: {e})")

    combined_markdown = "\n\n---\n\n".join(combined_markdown_parts)

    # ‡∏õ‡πâ‡∏≠‡∏á‡∏Å‡∏±‡∏ô‡∏¢‡∏≤‡∏ß‡πÄ‡∏Å‡∏¥‡∏ô‡πÑ‡∏õ
    if len(combined_markdown) > 20000:
        combined_markdown = combined_markdown[:20000]

    # ‡πÄ‡∏£‡∏µ‡∏¢‡∏Å LLM
    summary = call_llm_for_summary(
        content=combined_markdown,
        extra_instruction=extra_instruction,
        max_tokens=1500,
    )

    return summary
```

```python
    header = ws.row_values(1)
```
